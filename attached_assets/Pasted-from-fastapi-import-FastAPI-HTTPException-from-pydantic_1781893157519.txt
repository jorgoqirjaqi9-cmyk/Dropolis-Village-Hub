from fastapi import FastAPI, HTTPException
from pydantic import BaseModel
from typing import Dict, Optional
import uuid

app = FastAPI(title="Arena API")

# In-memory storage (use database in production)
arenas: Dict[str, dict] = {}

# Models
class CreateArena(BaseModel):
    name: str
    max_players: int = 2

class JoinArena(BaseModel):
    player_name: str

# Create Arena
@app.post("/arena/create")
def create_arena(data: CreateArena):
    arena_id = str(uuid.uuid4())[:8]
    arenas[arena_id] = {
        "id": arena_id,
        "name": data.name,
        "players": [],
        "max_players": data.max_players,
        "status": "waiting"
    }
    return {"arena_id": arena_id, "message": "Arena created"}

# Join Arena
@app.post("/arena/{arena_id}/join")
def join_arena(arena_id: str, data: JoinArena):
    if arena_id not in arenas:
        raise HTTPException(status_code=404, detail="Arena not found")
    
    arena = arenas[arena_id]
    
    if len(arena["players"]) >= arena["max_players"]:
        raise HTTPException(status_code=400, detail="Arena is full")
    
    arena["players"].append(data.player_name)
    
    if len(arena["players"]) == arena["max_players"]:
        arena["status"] = "ready"
    
    return {"message": f"{data.player_name} joined", "arena": arena}

# Get Arena Status
@app.get("/arena/{arena_id}")
def get_arena(arena_id: str):
    if arena_id not in arenas:
        raise HTTPException(status_code=404, detail="Arena not found")
    return arenas[arena_id]

# Start Match
@app.post("/arena/{arena_id}/start")
def start_match(arena_id: str):
    if arena_id not in arenas:
        raise HTTPException(status_code=404, detail="Arena not found")
    
    arena = arenas[arena_id]
    if len(arena["players"]) < 2:
        raise HTTPException(status_code=400, detail="Need at least 2 players")
    
    arena["status"] = "in_progress"
    return {"message": "Match started", "arena": arena}

# List All Arenas
@app.get("/arenas")
def list_arenas():
    return list(arenas.values())
